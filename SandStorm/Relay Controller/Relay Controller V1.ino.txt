/*
  Sandstorm T-Display S3 Relay Controller - FULL UPDATED VERSION

  Includes:
  - Cloud polling from HTTPS server
  - Fresh 30-second buffer support
  - Manual relay ON/OFF commands from Relay Monitor
  - All Relays Off command
  - Timeline relay playback
  - Startup safety: relays OFF and waits for server command
  - Does not require browser to POST directly to local ESP

  Server queue URL:
  https://icssolution.net/Sandstorm/api/esp_queue.php?device=tdisplay1

  Arduino libraries:
  - WiFi
  - WiFiClientSecure
  - HTTPClient
  - ArduinoJson
  - TFT_eSPI

  Important:
  The relay pins below are examples. For a real 16-relay build, an I/O expander
  such as MCP23017, PCF8575, or 74HC595 is strongly recommended.
*/

#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <TFT_eSPI.h>

// =======================
// EDIT THESE
// =======================
const char* WIFI_SSID = "House wifi";
const char* WIFI_PASS = "dkh45553";

const char* DEVICE_NAME = "tdisplay1";
const char* QUEUE_URL = "https://icssolution.net/Sandstorm/api/esp_queue.php?device=tdisplay1";

// Most SainSmart relay boards are active LOW.
#define RELAY_ACTIVE_LOW true

// Poll server every second.
const unsigned long POLL_INTERVAL_MS = 1000;

// Maximum timeline triggers.
#define MAX_TRIGGERS 500

// =======================
// RELAY PINS - EXAMPLES ONLY
// =======================
// Change these to match your wiring.
// T-Display S3 does not expose 16 ideal GPIO pins while using the display.
// Strongly consider using MCP23017/PCF8575/74HC595 for the relay board.
const int relayPins[16] = {
  1, 2, 3, 10,
  11, 12, 13, 14,
  15, 16, 17, 18,
  43, 44, 21, 38
};

// =======================
// DATA STRUCTURES
// =======================
struct Trigger {
  uint32_t timeMs;
  uint8_t relay;
  bool state;
  bool fired;
};

Trigger triggers[MAX_TRIGGERS];
int triggerCount = 0;

String lastCommandId = "";
String sessionName = "";

uint64_t payloadStartEpochMs = 0;
uint64_t payloadServerEpochMs = 0;

uint32_t localStartAtMs = 0;
uint32_t localTimelineStartMs = 0;
uint32_t armedDelayMs = 0;

bool timelineArmed = false;
bool timelineRunning = false;
bool timelineFinished = false;

unsigned long lastPollMs = 0;
unsigned long lastScreenMs = 0;

TFT_eSPI tft = TFT_eSPI();

// =======================
// DISPLAY
// =======================
void drawStatus(String line1, String line2 = "", String line3 = "") {
  tft.fillScreen(TFT_BLACK);
  tft.setTextColor(TFT_WHITE, TFT_BLACK);

  tft.setTextSize(2);
  tft.setCursor(8, 8);
  tft.println("Sandstorm Relay");

  tft.drawFastHLine(0, 34, 320, TFT_DARKGREY);

  tft.setCursor(8, 48);
  tft.println(line1);

  tft.setTextSize(1);
  tft.setCursor(8, 84);
  tft.println(line2);

  tft.setCursor(8, 104);
  tft.println(line3);

  tft.setCursor(8, 136);
  tft.print("IP: ");
  tft.println(WiFi.localIP());

  Serial.println("SCREEN: " + line1 + " | " + line2 + " | " + line3);
}

void drawCountdown() {
  if (millis() - lastScreenMs < 200) return;
  lastScreenMs = millis();

  int32_t leftMs = (int32_t)(localStartAtMs - millis());
  if (leftMs < 0) leftMs = 0;

  tft.fillRect(0, 42, 320, 92, TFT_BLACK);

  tft.setTextColor(TFT_YELLOW, TFT_BLACK);
  tft.setTextSize(2);
  tft.setCursor(8, 48);
  tft.println("ARMED");

  tft.setTextColor(TFT_WHITE, TFT_BLACK);
  tft.setCursor(8, 78);
  tft.print("Start in ");
  tft.print(leftMs / 1000.0, 1);
  tft.println("s");

  tft.setCursor(8, 108);
  tft.print("Triggers: ");
  tft.println(triggerCount);
}

void drawRunning(uint32_t elapsedMs) {
  if (millis() - lastScreenMs < 250) return;
  lastScreenMs = millis();

  int fired = 0;
  for (int i = 0; i < triggerCount; i++) {
    if (triggers[i].fired) fired++;
  }

  tft.fillRect(0, 42, 320, 92, TFT_BLACK);

  tft.setTextColor(TFT_GREEN, TFT_BLACK);
  tft.setTextSize(2);
  tft.setCursor(8, 48);
  tft.println("RUNNING");

  tft.setTextColor(TFT_WHITE, TFT_BLACK);
  tft.setCursor(8, 78);
  tft.print("Time: ");
  tft.print(elapsedMs / 1000.0, 3);
  tft.println("s");

  tft.setCursor(8, 108);
  tft.print("Fired: ");
  tft.print(fired);
  tft.print("/");
  tft.println(triggerCount);
}

// =======================
// RELAYS
// =======================
void setRelay(uint8_t relayNumber, bool on) {
  if (relayNumber < 1 || relayNumber > 16) return;

  int pin = relayPins[relayNumber - 1];

  if (RELAY_ACTIVE_LOW) {
    digitalWrite(pin, on ? LOW : HIGH);
  } else {
    digitalWrite(pin, on ? HIGH : LOW);
  }

  Serial.printf("Relay %u -> %s\n", relayNumber, on ? "ON" : "OFF");
}

void allRelaysOff() {
  for (int i = 1; i <= 16; i++) {
    setRelay(i, false);
  }
}

void setupRelays() {
  for (int i = 0; i < 16; i++) {
    pinMode(relayPins[i], OUTPUT);
  }
  allRelaysOff();
}

// =======================
// TIMELINE HELPERS
// =======================
void clearTimeline() {
  triggerCount = 0;
  timelineArmed = false;
  timelineRunning = false;
  timelineFinished = false;
  localStartAtMs = 0;
  localTimelineStartMs = 0;
  armedDelayMs = 0;
}

void sortTriggers() {
  for (int i = 0; i < triggerCount - 1; i++) {
    for (int j = i + 1; j < triggerCount; j++) {
      if (triggers[j].timeMs < triggers[i].timeMs) {
        Trigger tmp = triggers[i];
        triggers[i] = triggers[j];
        triggers[j] = tmp;
      }
    }
  }
}

// =======================
// MANUAL COMMANDS
// =======================
bool handleManualRelayCommand(JsonObject payload) {
  String payloadType = payload["type"] | "";
  if (payloadType != "manual_relay") return false;

  String command = payload["command"] | "relay";

  // Stop any active timeline when manual command is received.
  timelineArmed = false;
  timelineRunning = false;
  timelineFinished = false;

  if (command == "all_off") {
    allRelaysOff();
    drawStatus("MANUAL", "All relays OFF", "From server");
    Serial.println("Manual command: all relays OFF");
    return true;
  }

  if (command == "relay") {
    uint8_t relay = payload["relay"] | 0;
    bool state = ((int)(payload["state"] | 0)) == 1;

    if (relay >= 1 && relay <= 16) {
      setRelay(relay, state);
      drawStatus(
        "MANUAL",
        "Relay " + String(relay) + (state ? " ON" : " OFF"),
        "From server"
      );

      Serial.printf("Manual command: relay %u %s\n", relay, state ? "ON" : "OFF");
      return true;
    }

    drawStatus("MANUAL ERROR", "Invalid relay", String(relay));
    Serial.println("Manual command error: invalid relay");
    return true;
  }

  drawStatus("MANUAL ERROR", "Unknown command", command);
  Serial.println("Manual command error: unknown command");
  return true;
}

// =======================
// LOAD PAYLOAD
// =======================
bool loadPayload(JsonObject payload) {
  if (payload.isNull()) {
    Serial.println("Payload null");
    return false;
  }

  // Manual relay commands are handled immediately and do not arm a timeline.
  if (handleManualRelayCommand(payload)) {
    return true;
  }

  String payloadType = payload["type"] | "";
  if (payloadType != "" && payloadType != "relay_timeline") {
    Serial.println("Unknown payload type: " + payloadType);
    drawStatus("WAITING", "Unknown payload type", payloadType);
    return false;
  }

  sessionName = payload["session"] | "";
  payloadStartEpochMs = payload["start_epoch_ms"] | 0ULL;
  payloadServerEpochMs = payload["server_epoch_ms"] | 0ULL;

  if (payloadServerEpochMs == 0) {
    payloadServerEpochMs = payload["queued_at_ms"] | 0ULL;
  }

  if (payloadStartEpochMs == 0 || payloadServerEpochMs == 0) {
    Serial.println("Missing start_epoch_ms or server_epoch_ms");
    drawStatus("WAITING", "Bad payload time", "Send Live again");
    return false;
  }

  int64_t delaySigned = (int64_t)payloadStartEpochMs - (int64_t)payloadServerEpochMs;

  // Correct command should be about 30000ms. If a bad/stale command arrives,
  // force a 30 second safety delay rather than starting immediately.
  if (delaySigned < 5000) {
    Serial.println("Payload delay under 5 seconds. Forcing 30 second safety buffer.");
    delaySigned = 30000;
  }

  if (delaySigned > 120000) {
    delaySigned = 120000;
  }

  armedDelayMs = (uint32_t)delaySigned;
  localStartAtMs = millis() + armedDelayMs;

  JsonArray arr = payload["triggers"].as<JsonArray>();
  if (arr.isNull()) {
    Serial.println("Missing triggers");
    drawStatus("WAITING", "Missing triggers", "Send Live again");
    return false;
  }

  clearTimeline();

  // Restore values cleared above.
  armedDelayMs = (uint32_t)delaySigned;
  localStartAtMs = millis() + armedDelayMs;

  for (JsonObject t : arr) {
    if (triggerCount >= MAX_TRIGGERS) break;

    uint8_t relay = t["relay"] | 0;
    if (relay < 1 || relay > 16) continue;

    triggers[triggerCount].timeMs = t["time_ms"] | 0;
    triggers[triggerCount].relay = relay;
    triggers[triggerCount].state = ((int)(t["state"] | 0)) == 1;
    triggers[triggerCount].fired = false;

    Serial.printf(
      "Trigger %d: %lu ms relay %u state %d\n",
      triggerCount,
      (unsigned long)triggers[triggerCount].timeMs,
      triggers[triggerCount].relay,
      triggers[triggerCount].state
    );

    triggerCount++;
  }

  if (triggerCount == 0) {
    Serial.println("No valid triggers");
    drawStatus("WAITING", "No triggers loaded", "Add triggers/send again");
    return false;
  }

  sortTriggers();
  allRelaysOff();

  timelineArmed = true;
  timelineRunning = false;
  timelineFinished = false;

  Serial.print("Armed delay ms: ");
  Serial.println(armedDelayMs);

  drawStatus(
    "ARMED",
    "Buffer: " + String(armedDelayMs / 1000.0, 1) + " sec",
    "Triggers: " + String(triggerCount)
  );

  return true;
}

// =======================
// CLOUD POLLING
// =======================
void pollServer() {
  if (millis() - lastPollMs < POLL_INTERVAL_MS) return;
  lastPollMs = millis();

  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("WiFi disconnected");
    return;
  }

  WiFiClientSecure client;
  client.setInsecure();

  HTTPClient http;
  http.setTimeout(8000);

  if (!http.begin(client, QUEUE_URL)) {
    Serial.println("HTTP begin failed");
    return;
  }

  int code = http.GET();

  if (code != 200) {
    Serial.printf("Queue GET failed: %d\n", code);
    http.end();
    return;
  }

  String body = http.getString();
  http.end();

  JsonDocument doc;
  DeserializationError err = deserializeJson(doc, body);

  if (err) {
    Serial.print("JSON parse error: ");
    Serial.println(err.c_str());
    return;
  }

  bool hasCommand = doc["has_command"] | false;

  if (!hasCommand) {
    if (!timelineArmed && !timelineRunning) {
      static unsigned long lastIdleDraw = 0;
      if (millis() - lastIdleDraw > 5000) {
        lastIdleDraw = millis();
        drawStatus("READY", "Waiting for server", DEVICE_NAME);
      }
    }
    return;
  }

  String commandId = doc["command_id"] | "";
  if (commandId == "") return;

  if (commandId == lastCommandId) {
    return;
  }

  JsonObject payload = doc["payload"].as<JsonObject>();
  bool ok = loadPayload(payload);

  // Only mark the command complete after it loaded successfully.
  if (ok) {
    lastCommandId = commandId;
    Serial.println("Accepted command: " + lastCommandId);
  } else {
    Serial.println("Payload failed; command not accepted");
  }
}

// =======================
// TIMELINE ENGINE
// =======================
void runTimeline() {
  if (timelineArmed && !timelineRunning) {
    drawCountdown();

    int32_t leftMs = (int32_t)(localStartAtMs - millis());

    if (leftMs <= 0) {
      timelineArmed = false;
      timelineRunning = true;
      timelineFinished = false;
      localTimelineStartMs = millis();

      for (int i = 0; i < triggerCount; i++) {
        triggers[i].fired = false;
      }

      allRelaysOff();

      drawStatus("STARTED", "Timeline running", "Triggers: " + String(triggerCount));
      Serial.println("Timeline started after buffer");
    }

    return;
  }

  if (timelineRunning) {
    uint32_t elapsedMs = millis() - localTimelineStartMs;

    for (int i = 0; i < triggerCount; i++) {
      if (!triggers[i].fired && elapsedMs >= triggers[i].timeMs) {
        setRelay(triggers[i].relay, triggers[i].state);
        triggers[i].fired = true;
      }
    }

    drawRunning(elapsedMs);

    if (triggerCount > 0 && elapsedMs > triggers[triggerCount - 1].timeMs + 2000) {
      timelineRunning = false;
      timelineFinished = true;
      drawStatus("FINISHED", "Timeline complete", "Relays keep last state");
      Serial.println("Timeline finished");
    }
  }
}

// =======================
// SETUP / LOOP
// =======================
void setup() {
  Serial.begin(115200);
  delay(500);

  tft.init();
  tft.setRotation(1);
  tft.fillScreen(TFT_BLACK);

  setupRelays();

  drawStatus("WiFi connecting", WIFI_SSID);

  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  int tries = 0;
  while (WiFi.status() != WL_CONNECTED && tries < 60) {
    delay(500);
    Serial.print(".");
    tries++;
  }
  Serial.println();

  if (WiFi.status() != WL_CONNECTED) {
    drawStatus("WiFi FAILED", "Check SSID/password");
    Serial.println("WiFi failed");
    return;
  }

  clearTimeline();
  allRelaysOff();

  drawStatus("READY", "Waiting for server", DEVICE_NAME);

  Serial.println("Ready.");
  Serial.print("IP: ");
  Serial.println(WiFi.localIP());
  Serial.println("Queue URL:");
  Serial.println(QUEUE_URL);
}

void loop() {
  pollServer();
  runTimeline();
}
